499 /* Cordio requires the call to DmConnAccept to properly catch direct advertising timeout.
500 * This call returns a connection ID and expect the peer address and the peer address type.
501 * This PAL following HCI expectations, these information are not present when the higher
502 * level calls the function to enable advertising.
503 * This data structure caches these informations and use them in case of direct advertising.
504 * Similarly, direct advertising timeout doesn't trigger an advertising timeout instead it
505 * triggers a disconnection with the status 60. Information on running direct advertising
506 * is stored here.
507 */
508 struct direct_adv_cb_t {
509 direct_adv_cb_t() :
510 peer_address_type(advertising_peer_address_type_t::PUBLIC),
511 state(free)
512 {
513 }
514
515 address_t peer_address;
516 advertising_peer_address_type_t peer_address_type;
517 advertising_handle_t advertising_handle;
518 connection_handle_t connection_handle;
519 bool low_duty_cycle;
520 enum {
521 free = 0,
522 pending,
523 running
524 } state;
525 };

552 static uint8_t create_all_phys_value(
553 const phy_set_t &tx_phys,
554 const phy_set_t &rx_phys
555 )
556 {
557 /* if phy set is empty set corresponding all_phys bit to 1 */
558 uint8_t all_phys = 0;
559 if (tx_phys.value() == 0) {
560 all_phys |= 0x01;
561 }
562 if (rx_phys.value() == 0) {
563 all_phys |= 0x02;
564 }
565 return all_phys;
566 }
