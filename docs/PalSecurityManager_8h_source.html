41class KeyDistribution {
42public:
43 enum KeyDistributionFlags_t {
44 KEY_DISTRIBUTION_NONE = 0x00,
45 KEY_DISTRIBUTION_ENCRYPTION = 0x01,
46 KEY_DISTRIBUTION_IDENTITY = 0x02,
47 KEY_DISTRIBUTION_SIGNING = 0x04,
48 KEY_DISTRIBUTION_LINK = 0x08,
49 KEY_DISTRIBUTION_ALL = 0x0F
50 };
51
52 KeyDistribution() : _value(0) { }
53 KeyDistribution(uint8_t value) : _value(value) { }
54 KeyDistribution(bool encryption,
55 bool identity,
56 bool signing,
57 bool link) : _value(0) {
58 set_encryption(encryption);
59 set_identity(identity);
60 set_signing(signing);
61 set_link(link);
62 }
63
64 bool get_encryption() const {
65 return _value & KEY_DISTRIBUTION_ENCRYPTION;
66 }
67 bool get_identity() const {
68 return _value & KEY_DISTRIBUTION_IDENTITY;
69 }
70 bool get_signing() const {
71 return _value & KEY_DISTRIBUTION_SIGNING;
72 }
73 bool get_link() const {
74 return _value & KEY_DISTRIBUTION_LINK;
75 }
76
77 void set_encryption(bool enabled = true) {
78 if (enabled) {
79 _value |= KEY_DISTRIBUTION_ENCRYPTION;
80 } else {
81 _value &= ~KEY_DISTRIBUTION_ENCRYPTION;
82 }
83 }
84 void set_identity(bool enabled = true) {
85 if (enabled) {
86 _value |= KEY_DISTRIBUTION_IDENTITY;
87 } else {
88 _value &= ~KEY_DISTRIBUTION_IDENTITY;
89 }
90 }
91 void set_signing(bool enabled = true) {
92 if (enabled) {
93 _value |= KEY_DISTRIBUTION_SIGNING;
94 } else {
95 _value &= ~KEY_DISTRIBUTION_SIGNING;
96 }
97 }
98 void set_link(bool enabled = true) {
99 if (enabled) {
100 _value |= KEY_DISTRIBUTION_LINK;
101 } else {
102 _value &= ~KEY_DISTRIBUTION_LINK;
103 }
104 }
105
106 operator uint8_t() {
107 return _value;
108 }
109
110 KeyDistribution operator&(const KeyDistribution& other) const {
111 KeyDistribution result(this->value() & other.value());
112 return result;
113 }
114
115 KeyDistribution& operator&=(const KeyDistribution& other) {
116 this->_value = this->_value & other.value();
117 return *this;
118 }
119
120 uint8_t value() const {
121 return _value;
122 }
123
124private:
125 uint8_t _value;
126};

132class AuthenticationMask {
133public:
134 enum AuthenticationFlags_t {
135 AUTHENTICATION_BONDABLE = 0x01,
136 AUTHENTICATION_MITM = 0x04, /* 0x02 missing because bonding uses two bits */
137 AUTHENTICATION_SECURE_CONNECTIONS = 0x08,
138 AUTHENTICATION_KEYPRESS_NOTIFICATION = 0x10
139 };
140
141 AuthenticationMask() : _value(0) { }
142 AuthenticationMask(uint8_t value) : _value(value) { }
143 AuthenticationMask(bool bondable,
144 bool mitm,
145 bool secure_connections,
146 bool keypress) : _value(0) {
147 set_bondable(bondable);
148 set_mitm(mitm);
149 set_secure_connections(secure_connections);
150 set_keypress_notification(keypress);
151 }
152
153 bool get_bondable() const {
154 return _value & AUTHENTICATION_BONDABLE;
155 }
156 bool get_mitm() const {
157 return _value & AUTHENTICATION_MITM;
158 }
159 bool get_secure_connections() const {
160 return _value & AUTHENTICATION_SECURE_CONNECTIONS;
161 }
162 bool get_keypress_notification() const {
163 return _value & AUTHENTICATION_KEYPRESS_NOTIFICATION;
164 }
165
166 void set_bondable(bool enabled = true) {
167 if (enabled) {
168 _value |= AUTHENTICATION_BONDABLE;
169 } else {
170 _value &= ~AUTHENTICATION_BONDABLE;
171 }
172 }
173 void set_mitm(bool enabled = true) {
174 if (enabled) {
175 _value |= AUTHENTICATION_MITM;
176 } else {
177 _value &= ~AUTHENTICATION_MITM;
178 }
179 }
180 void set_secure_connections(bool enabled = true) {
181 if (enabled) {
182 _value |= AUTHENTICATION_SECURE_CONNECTIONS;
183 } else {
184 _value &= ~AUTHENTICATION_SECURE_CONNECTIONS;
185 }
186 }
187 void set_keypress_notification(bool enabled = true) {
188 if (enabled) {
189 _value |= AUTHENTICATION_KEYPRESS_NOTIFICATION;
190 } else {
191 _value &= ~AUTHENTICATION_KEYPRESS_NOTIFICATION;
192 }
193 }
194
195 operator uint8_t() {
196 return _value;
197 }
198 uint8_t value() const {
199 return _value;
200 }
201
202private:
203 uint8_t _value;
204};
