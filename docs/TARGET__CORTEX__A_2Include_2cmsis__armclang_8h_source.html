174__STATIC_FORCEINLINE uint32_t __ROR(uint32_t op1, uint32_t op2)
175{
176 op2 %= 32U;
177 if (op2 == 0U)
178 {
179 return op1;
180 }
181 return (op1 >> op2) | (op1 << (32U - op2));
182}

204__STATIC_FORCEINLINE uint8_t __CLZ(uint32_t value)
205{
206 /* Even though __builtin_clz produces a CLZ instruction on ARM, formally
207 __builtin_clz(0) is undefined behaviour, so handle this case specially.
208 This guarantees ARM-compatible results if happening to compile on a non-ARM
209 target, and ensures the compiler doesn't decide to activate any
210 optimisations using the logic "value was passed to __builtin_clz, so it
211 is non-zero".
212 ARM Compiler 6.10 and possibly earlier will optimise this test away, leaving a
213 single CLZ instruction.
214 */
215 if (value == 0U)
216 {
217 return 32U;
218 }
219 return __builtin_clz(value);
220}

552__STATIC_INLINE void __FPU_Enable(void)
553{
554 __ASM volatile(
555 //Permit access to VFP/NEON, registers by modifying CPACR
556 " MRC p15,0,R1,c1,c0,2 \n"
557 " ORR R1,R1,#0x00F00000 \n"
558 " MCR p15,0,R1,c1,c0,2 \n"
559
560 //Ensure that subsequent instructions occur in the context of VFP/NEON access permitted
561 " ISB \n"
562
563 //Enable VFP/NEON
564 " VMRS R1,FPEXC \n"
565 " ORR R1,R1,#0x40000000 \n"
566 " VMSR FPEXC,R1 \n"
567
568 //Initialise VFP/NEON registers to 0
569 " MOV R2,#0 \n"
570
571 //Initialise D16 registers to 0
572 " VMOV D0, R2,R2 \n"
573 " VMOV D1, R2,R2 \n"
574 " VMOV D2, R2,R2 \n"
575 " VMOV D3, R2,R2 \n"
576 " VMOV D4, R2,R2 \n"
577 " VMOV D5, R2,R2 \n"
578 " VMOV D6, R2,R2 \n"
579 " VMOV D7, R2,R2 \n"
580 " VMOV D8, R2,R2 \n"
581 " VMOV D9, R2,R2 \n"
582 " VMOV D10,R2,R2 \n"
583 " VMOV D11,R2,R2 \n"
584 " VMOV D12,R2,R2 \n"
585 " VMOV D13,R2,R2 \n"
586 " VMOV D14,R2,R2 \n"
587 " VMOV D15,R2,R2 \n"
588
589#if (defined(__ARM_NEON) && (__ARM_NEON == 1))
590 //Initialise D32 registers to 0
591 " VMOV D16,R2,R2 \n"
592 " VMOV D17,R2,R2 \n"
593 " VMOV D18,R2,R2 \n"
594 " VMOV D19,R2,R2 \n"
595 " VMOV D20,R2,R2 \n"
596 " VMOV D21,R2,R2 \n"
597 " VMOV D22,R2,R2 \n"
598 " VMOV D23,R2,R2 \n"
599 " VMOV D24,R2,R2 \n"
600 " VMOV D25,R2,R2 \n"
601 " VMOV D26,R2,R2 \n"
602 " VMOV D27,R2,R2 \n"
603 " VMOV D28,R2,R2 \n"
604 " VMOV D29,R2,R2 \n"
605 " VMOV D30,R2,R2 \n"
606 " VMOV D31,R2,R2 \n"
607#endif
608
609 //Initialise FPSCR to a known state
610 " VMRS R1,FPSCR \n"
611 " LDR R2,=0x00086060 \n" //Mask off all bits that do not have to be preserved. Non-preserved bits can/should be zero.
612 " AND R1,R1,R2 \n"
613 " VMSR FPSCR,R1 "
614 : : : "cc", "r1", "r2"
615 );
