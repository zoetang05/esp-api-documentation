141#if !defined(MBEDTLS_ECP_WINDOW_SIZE)
142/*
143 * Maximum "window" size used for point multiplication.
144 * Default: 6.
145 * Minimum value: 2. Maximum value: 7.
146 *
147 * Result is an array of at most ( 1 << ( MBEDTLS_ECP_WINDOW_SIZE - 1 ) )
148 * points used for point multiplication. This value is directly tied to EC
149 * peak memory usage, so decreasing it by one should roughly cut memory usage
150 * by two (if large curves are in use).
151 *
152 * Reduction in size may reduce speed, but larger curves are impacted first.
153 * Sample performances (in ECDHE handshakes/s, with FIXED_POINT_OPTIM = 1):
154 * w-size: 6 5 4 3 2
155 * 521 145 141 135 120 97
156 * 384 214 209 198 177 146
157 * 256 320 320 303 262 226
158 * 224 475 475 453 398 342
159 * 192 640 640 633 587 476
160 */
161#define MBEDTLS_ECP_WINDOW_SIZE 6
162#endif /* MBEDTLS_ECP_WINDOW_SIZE */
163
164#if !defined(MBEDTLS_ECP_FIXED_POINT_OPTIM)
165/*
166 * Trade memory for speed on fixed-point multiplication.
167 *
168 * This speeds up repeated multiplication of the generator (that is, the
169 * multiplication in ECDSA signatures, and half of the multiplications in
170 * ECDSA verification and ECDHE) by a factor roughly 3 to 4.
171 *
172 * The cost is increasing EC peak memory usage by a factor roughly 2.
173 *
174 * Change this value to 0 to reduce peak memory usage.
175 */
176#define MBEDTLS_ECP_FIXED_POINT_OPTIM 1
177#endif /* MBEDTLS_ECP_FIXED_POINT_OPTIM */
