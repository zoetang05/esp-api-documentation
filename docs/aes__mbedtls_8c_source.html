62#ifndef GET_UINT32_LE
63#define GET_UINT32_LE(n,b,i) \
64{ \
65 (n) = ( (uint32_t) (b)[(i) ] ) \
66 | ( (uint32_t) (b)[(i) + 1] << 8 ) \
67 | ( (uint32_t) (b)[(i) + 2] << 16 ) \
68 | ( (uint32_t) (b)[(i) + 3] << 24 ); \

72#ifndef PUT_UINT32_LE
73#define PUT_UINT32_LE(n,b,i) \
74{ \

77 (b)[(i) + 2] = (unsigned char) ( ( (n) >> 16 ) & 0xFF ); \

233/*
234 * AES key schedule (encryption)
235 */
236static int mbedtls_aes_setkey_enc(mbedtls_aes_context *ctx, const unsigned char *key,
237 unsigned int keysize)
238{
239 unsigned int i;
240 uint32_t *RK;
241
242 switch (keysize) {
243 case 128:
244 ctx->nr = 10;
245 break;
246 default :
247 return (MBEDTLS_ERR_AES_INVALID_KEY_LENGTH);
248 }
249
250 ctx->rk = RK = ctx->buf;
251
252 for (i = 0; i < (keysize >> 5); i++) {
253 GET_UINT32_LE(RK[i], key, i << 2);
254 }
255
256 switch (ctx->nr) {
257 case 10:
258
259 for (i = 0; i < 10; i++, RK += 4) {
260 RK[4] = RK[0] ^ RCON[i] ^
261 ((uint32_t) FSb[(RK[3] >> 8) & 0xFF ]) ^
262 ((uint32_t) FSb[(RK[3] >> 16) & 0xFF ] << 8) ^
263 ((uint32_t) FSb[(RK[3] >> 24) & 0xFF ] << 16) ^
264 ((uint32_t) FSb[(RK[3]) & 0xFF ] << 24);
265
266 RK[5] = RK[1] ^ RK[4];
267 RK[6] = RK[2] ^ RK[5];
268 RK[7] = RK[3] ^ RK[6];
269 }
270 break;
271 }
272
273 return (0);
274}

300/*
301 * AES-ECB block encryption/decryption
302 */
303static int mbedtls_aes_crypt_ecb(mbedtls_aes_context *ctx,
304 int mode,
305 const unsigned char input[16],
306 unsigned char output[16])
307{
308 int i;
309 uint32_t *RK, X0, X1, X2, X3, Y0, Y1, Y2, Y3;
310
311 RK = ctx->rk;
312
313 GET_UINT32_LE(X0, input, 0);
314 X0 ^= *RK++;
315 GET_UINT32_LE(X1, input, 4);
316 X1 ^= *RK++;
317 GET_UINT32_LE(X2, input, 8);
318 X2 ^= *RK++;
319 GET_UINT32_LE(X3, input, 12);
320 X3 ^= *RK++;
321
322 if (mode == MBEDTLS_AES_DECRYPT) {
323 return (MBEDTLS_ERR_AES_UNSUPPORTED);
324 } else { /* MBEDTLS_AES_ENCRYPT */
325 for (i = (ctx->nr >> 1) - 1; i > 0; i--) {
326 AES_FROUND(Y0, Y1, Y2, Y3, X0, X1, X2, X3);
327 AES_FROUND(X0, X1, X2, X3, Y0, Y1, Y2, Y3);
328 }
329
330 AES_FROUND(Y0, Y1, Y2, Y3, X0, X1, X2, X3);
331
332 X0 = *RK++ ^ \
333 ((uint32_t) FSb[(Y0) & 0xFF ]) ^
334 ((uint32_t) FSb[(Y1 >> 8) & 0xFF ] << 8) ^
335 ((uint32_t) FSb[(Y2 >> 16) & 0xFF ] << 16) ^
336 ((uint32_t) FSb[(Y3 >> 24) & 0xFF ] << 24);
337
338 X1 = *RK++ ^ \
339 ((uint32_t) FSb[(Y1) & 0xFF ]) ^
340 ((uint32_t) FSb[(Y2 >> 8) & 0xFF ] << 8) ^
341 ((uint32_t) FSb[(Y3 >> 16) & 0xFF ] << 16) ^
342 ((uint32_t) FSb[(Y0 >> 24) & 0xFF ] << 24);
343
344 X2 = *RK++ ^ \
345 ((uint32_t) FSb[(Y2) & 0xFF ]) ^
346 ((uint32_t) FSb[(Y3 >> 8) & 0xFF ] << 8) ^
347 ((uint32_t) FSb[(Y0 >> 16) & 0xFF ] << 16) ^
348 ((uint32_t) FSb[(Y1 >> 24) & 0xFF ] << 24);
349
350 X3 = *RK++ ^ \
351 ((uint32_t) FSb[(Y3) & 0xFF ]) ^
352 ((uint32_t) FSb[(Y0 >> 8) & 0xFF ] << 8) ^
353 ((uint32_t) FSb[(Y1 >> 16) & 0xFF ] << 16) ^
354 ((uint32_t) FSb[(Y2 >> 24) & 0xFF ] << 24);
355 }
356
357 PUT_UINT32_LE(X0, output, 0);
358 PUT_UINT32_LE(X1, output, 4);
359 PUT_UINT32_LE(X2, output, 8);
360 PUT_UINT32_LE(X3, output, 12);
361
362 return (0);
363}
