86/* 6LoWPAN Types
87 *
88 * Pattern Header Type
89 * +------------+-----------------------------------------------+
90 * | 00 xxxxxx | NALP - Not a LoWPAN frame |
91 * | 01 000000 | ESC - Additional Dispatch byte follows |
92 * | 01 000001 | IPv6 - Uncompressed IPv6 Addresses |
93 * | 01 000010 | LOWPAN_HC1 - LOWPAN_HC1 compressed IPv6 |
94 * | 01 000011 | LOWPAN_DFF - Depth-First Forwarding (RFC6971) |
95 * | 01 000100 | reserved - Reserved for future use |
96 * | ... | reserved - Reserved for future use |
97 * | 01 001111 | reserved - Reserved for future use |
98 * | 01 010000 | LOWPAN_BC0 - LOWPAN_BC0 broadcast |
99 * | 01 010001 | reserved - Reserved for future use |
100 * | ... | reserved - Reserved for future use |
101 * | 01 011111 | reserved - Reserved for future use |
102 * | 01 1xxxxx | LOWPAN_IPHC- RFC 6282 IPv6 Header Compression |
103 * | 10 xxxxxx | MESH - Mesh Header |
104 * | 11 000xxx | FRAG1 - Fragmentation Header (first) |
105 * | 11 001000 | reserved - Reserved for future use |
106 * | ... | reserved - Reserved for future use |
107 * | 11 011111 | reserved - Reserved for future use |
108 * | 11 100xxx | FRAGN - Fragmentation Header (subsequent)|
109 * | 11 101000 | reserved - Reserved for future use |
110 * | ... | reserved - Reserved for future use |
111 * | 11 111111 | reserved - Reserved for future use |
112 * +------------+-----------------------------------------------+
113 *
114 */
115#define LOWPAN_NALP 0x00 /* 00 xxxxxx Not LoWPAN */
116#define LOWPAN_NALP_MASK 0xC0
117#define LOWPAN_DISPATCH_ESC 0x40 /* RFC 6282 supercedes RFC 4944 */
118#define LOWPAN_DISPATCH_IPV6 0x41
119#define LOWPAN_DISPATCH_HC1 0x42
120#define LOWPAN_DISPATCH_DFF 0x43 /* RFC 6971 */
121#define LOWPAN_DISPATCH_BC0 0x50
122#define LOWPAN_DISPATCH_IPHC 0x60 /* 01 1xxxxx IPHC */
123#define LOWPAN_DISPATCH_IPHC_MASK 0xE0
124#define LOWPAN_MESH 0x80 /* 10 xxxxxx MESH */
125#define LOWPAN_MESH_MASK 0xC0
126#define LOWPAN_FRAG1 0xC0 /* 11 000xxx FRAG1 */
127#define LOWPAN_FRAG1_MASK 0xF8
128#define LOWPAN_FRAGN 0xE0 /* 11 100xxx FRAGN */
129#define LOWPAN_FRAGN_MASK 0xF8
