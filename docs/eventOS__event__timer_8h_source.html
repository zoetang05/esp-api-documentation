39static inline uint32_t eventOS_event_timer_ticks_to_ms(uint32_t ticks)
40{
41 NS_STATIC_ASSERT(1000 % EVENTOS_EVENT_TIMER_HZ == 0, "Assuming whole number of ms per tick")
42 return ticks * (1000 / EVENTOS_EVENT_TIMER_HZ);
43}

45/* Convert ms to ticks, rounding up (so 9ms = 1 tick, 10ms = 1 tick, 11ms = 2 ticks) */
46static inline uint32_t eventOS_event_timer_ms_to_ticks(uint32_t ms)
47{
48 NS_STATIC_ASSERT(1000 % EVENTOS_EVENT_TIMER_HZ == 0, "Assuming whole number of ms per tick")
49 return (ms + (1000 / EVENTOS_EVENT_TIMER_HZ) - 1) / (1000 / EVENTOS_EVENT_TIMER_HZ);
50}
