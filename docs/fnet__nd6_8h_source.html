40/**************************************************************
41* RFC4861. 10. Protocol Constants.
42***************************************************************
43* Host constants:
44* MAX_RTR_SOLICITATION_DELAY 1 second
45* RTR_SOLICITATION_INTERVAL 4 seconds
46* MAX_RTR_SOLICITATIONS 3 transmissions
47* Node constants:
48* MAX_MULTICAST_SOLICIT 3 transmissions
49* MAX_UNICAST_SOLICIT 3 transmissions
50* MAX_ANYCAST_DELAY_TIME 1 second
51* MAX_NEIGHBOR_ADVERTISEMENT 3 transmissions
52* REACHABLE_TIME 30,000 milliseconds
53* RETRANS_TIMER 1,000 milliseconds
54* DELAY_FIRST_PROBE_TIME 5 seconds
55* MIN_RANDOM_FACTOR .5
56* MAX_RANDOM_FACTOR 1.5
57***************************************************************/

59/* If a host sends MAX_RTR_SOLICITATIONS solicitations, and receives no
60 * Router Advertisements after having waited MAX_RTR_SOLICITATION_DELAY
61 * seconds after sending the last solicitation, the host concludes that
62 * there are no routers on the link for the purpose of [ADDRCONF].
63 * However, the host continues to receive and process Router
64 * Advertisements messages in the event that routers appear on the link.
65 *
66 * A host SHOULD transmit up to MAX_RTR_SOLICITATIONS Router
67 * Solicitation messages, each separated by at least
68 * RTR_SOLICITATION_INTERVAL seconds.
69 */
70#define FNET_ND6_MAX_RTR_SOLICITATIONS (3U) /* transmissions */
71#define FNET_ND6_MAX_RTR_SOLICITATION_DELAY (1000U) /* ms */
72#define FNET_ND6_RTR_SOLICITATION_INTERVAL (4000U) /* ms */
73
74/* If no Neighbor Advertisement is received after MAX_MULTICAST_SOLICIT
75* solicitations, address resolution has failed. The sender MUST return
76* ICMP destination unreachable indications with code 3 (Address
77* Unreachable) for each packet queued awaiting address resolution.
78*/
79#define FNET_ND6_MAX_MULTICAST_SOLICIT (3U) /* transmissions */
80
81/*
82 * Default value of the time between retransmissions of Neighbor
83 * Solicitation messages to a neighbor when
84 * resolving the address or when probing the
85 * reachability of a neighbor. Also used during Duplicate
86 * Address Detection (RFC4862).
87 */
88#define FNET_ND6_RETRANS_TIMER (1000U) /* ms */
89
90/*
91 * Default value of the time a neighbor is considered reachable after
92 * receiving a reachability confirmation.
93 *
94 * This value should be a uniformly distributed
95 * random value between MIN_RANDOM_FACTOR and
96 * MAX_RANDOM_FACTOR times BaseReachableTime
97 * milliseconds. A new random value should be
98 * calculated when BaseReachableTime changes (due to
99 * Router Advertisements) or at least every few
100 * hours even if
101 */
102#define FNET_ND6_REACHABLE_TIME (30000U) /* ms */
103
104/*
105 * If no reachability confirmation is received
106 * within DELAY_FIRST_PROBE_TIME seconds of entering the
107 * DELAY state, send a Neighbor Solicitation and change
108 * the state to PROBE.
109 */
110#define FNET_ND6_DELAY_FIRST_PROBE_TIME (5000U) /*ms*/
111
112/*
113 * If no response is
114 * received after waiting RetransTimer milliseconds after sending the
115 * MAX_UNICAST_SOLICIT solicitations, retransmissions cease and the
116 * entry SHOULD be deleted.
117 */
118#define FNET_ND6_MAX_UNICAST_SOLICIT (3U) /*times*/

160/**************************************************************
161* Neighbor’s reachability states, based on RFC4861.
162**************************************************************/
163typedef enum fnet_nd6_neighbor_state
164{
165 FNET_ND6_NEIGHBOR_STATE_NOTUSED = 0, /* The entry is not used - free.*/
166 FNET_ND6_NEIGHBOR_STATE_INCOMPLETE = 1, /* Address resolution is in progress and the link-layer
167 * address of the neighbor has not yet been determined.*/
168 FNET_ND6_NEIGHBOR_STATE_REACHABLE = 2, /* Roughly speaking, the neighbor is known to have been
169 * reachable recently (within tens of seconds ago).*/
170 FNET_ND6_NEIGHBOR_STATE_STALE = 3, /* The neighbor is no longer known to be reachable but
171 * until traffic is sent to the neighbor, no attempt
172 * should be made to verify its reachability.*/
173 FNET_ND6_NEIGHBOR_STATE_DELAY = 4, /* The neighbor is no longer known to be reachable, and
174 * traffic has recently been sent to the neighbor.
175 * Rather than probe the neighbor immediately, however,
176 * delay sending probes for a short while in order to
177 * give upper-layer protocols a chance to provide
178 * reachability confirmation.*/
179 FNET_ND6_NEIGHBOR_STATE_PROBE = 5 /* The neighbor is no longer known to be reachable, and
180 * unicast Neighbor Solicitation probes are being sent to
181 * verify reachability.*/
182} fnet_nd6_neighbor_state_t;
