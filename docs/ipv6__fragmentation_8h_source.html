25/* How large a fragmented packet are we prepared to handle? */
26/* Must be a multiple of 8. This default can be overridden. */
27#ifndef IPV6_FRAG_MRU
28#define IPV6_FRAG_MRU 1504
29#endif
30
31#if IPV6_FRAG_MRU & 7
32#error IPV6_FRAG_MRU must be a multiple of 8
33#endif
34
35/* Note that the MRU above applies only to fragmented packets. RFC 2460 says:
36 *
37 * From each link to which a node is directly attached, the node must be
38 * able to accept packets as large as that link's MTU.
39 *
40 * The IP spec doesn't seem to be explicit, but it is generally assumed that the
41 * MRU should be greater than or equal to the MTU of every link - we shouldn't
42 * refuse a packet just because it was fragmented.
43 */
44
45#if IPV6_FRAG_MRU < IPV6_MIN_FRAG_MRU
46#error "IPV6_MRU too small"
47#endif
