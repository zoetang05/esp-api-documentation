145#define LWIP_DEBUGF(debug, message) do { \
146 if ( \
147 ((debug) & LWIP_DBG_ON) && \
148 ((debug) & LWIP_DBG_TYPES_ON) && \
149 ((s16_t)((debug) & LWIP_DBG_MASK_LEVEL) >= LWIP_DBG_MIN_LEVEL)) { \
150 LWIP_PLATFORM_DIAG(message); \
151 if ((debug) & LWIP_DBG_HALT) { \
152 while(1); \

155 } while(0)
