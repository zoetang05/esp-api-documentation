514#elif defined(__ICCARM__)
515#define MBED_FILENAME (strrchr(__FILE__, '/') ? strrchr(__FILE__, '/') + 1 : strrchr(__FILE__, '\\') ? strrchr(__FILE__, '\\') + 1 : __FILE__)

517#define MBED_FILENAME __FILE__
518#endif
519#endif // #ifndef MBED_FILENAME
