622NS_LIST_FN void ns_list_add_to_start_(ns_list_t *list, ns_list_offset_t offset, void *restrict entry)
623{
624 void *next;
625
626 NS_LIST_PREV_(entry, offset) = &list->first_entry;
627 NS_LIST_NEXT_(entry, offset) = next = list->first_entry;
628
629 if (next) {
630 NS_LIST_PREV_(next, offset) = &NS_LIST_NEXT_(entry, offset);
631 } else {
632 list->last_nextptr = &NS_LIST_NEXT_(entry, offset);
633 }
634
635 list->first_entry = entry;
636}

654NS_LIST_FN void ns_list_add_before_(ns_list_offset_t offset, void *current, void *restrict entry)
655{
656 void **prev_nextptr;

658 NS_LIST_NEXT_(entry, offset) = current;

679NS_LIST_FN void *ns_list_get_previous_(const ns_list_t *list, ns_list_offset_t offset, const void *current)
680{
681 if (current == list->first_entry) {
682 return NULL;
683 }
684
685 // Tricky. We don't have a direct previous pointer, but a pointer to the
686 // pointer that points to us - ie &head->first_entry OR &{prev}->next.
687 // This makes life easier on insertion and removal, but this is where we
688 // pay the price.
689
690 // We have to check manually for being the first entry above, so we know it's
691 // a real link's next pointer. Then next is the first field of
692 // ns_list_link_t, so we can use the normal offset value.
693
694 return NS_LIST_ENTRY_(NS_LIST_PREV_(current, offset), offset);
695}

697NS_LIST_FN void *ns_list_get_last_(const ns_list_t *list, ns_list_offset_t offset)

699 if (!list->first_entry) {
700 return NULL;

707NS_LIST_FN void ns_list_remove_(ns_list_t *list, ns_list_offset_t offset, void *removed)
708{
709 void *next;
710 void **prev_nextptr;
711
712 next = NS_LIST_NEXT_(removed, offset);
713 prev_nextptr = NS_LIST_PREV_(removed, offset);
714 if (next) {
715 NS_LIST_PREV_(next, offset) = prev_nextptr;
716 } else {
717 list->last_nextptr = prev_nextptr;
718 }
719 *prev_nextptr = next;
720
721 ns_list_link_init_(NS_LIST_LINK_(removed, offset));
722}

724NS_LIST_FN void ns_list_replace_(ns_list_t *list, ns_list_offset_t offset, void *current, void *restrict replacement)
725{
726 void *next;
727 void **prev_nextptr;
728
729 NS_LIST_PREV_(replacement, offset) = prev_nextptr = NS_LIST_PREV_(current, offset);
730 NS_LIST_NEXT_(replacement, offset) = next = NS_LIST_NEXT_(current, offset);
731
732 if (next) {
733 NS_LIST_PREV_(next, offset) = &NS_LIST_NEXT_(replacement, offset);
734 } else {
735 list->last_nextptr = &NS_LIST_NEXT_(replacement, offset);
736 }
737 *prev_nextptr = replacement;
738
739 ns_list_link_init_(NS_LIST_LINK_(current, offset));
740}

742NS_LIST_FN void ns_list_concatenate_(ns_list_t *dst, ns_list_t *src, ns_list_offset_t offset)
743{
744 ns_list_link_t *src_first;
745
746 src_first = src->first_entry;
747 if (!src_first) {
748 return;
749 }
750
751 *dst->last_nextptr = src_first;
752 NS_LIST_PREV_(src_first, offset) = dst->last_nextptr;
753 dst->last_nextptr = src->last_nextptr;
754
755 ns_list_init_(src);
756}

758NS_LIST_FN uint_fast16_t ns_list_count_(const ns_list_t *list, ns_list_offset_t offset)
759{
760 uint_fast16_t count = 0;
761
762 for (void *p = list->first_entry; p; p = NS_LIST_NEXT_(p, offset)) {
763 count++;
764 }
765
766 return count;
