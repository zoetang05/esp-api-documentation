99/* Extensions to standard mbed TLS - output the first bits of a hash only */
100/* Number of bits must be a multiple of 32, and <=256 */
101static inline void ns_sha256_finish_nbits(ns_sha256_context *ctx, void *output, unsigned obits)
102{
103 if (obits == 256) {
104#if (MBEDTLS_VERSION_MAJOR >= 3)
105 (void)mbedtls_sha256_finish(ctx, output);
106#else
107 (void)mbedtls_sha256_finish_ret(ctx, output);
108#endif
109 } else {
110 uint8_t sha256[32];
111#if (MBEDTLS_VERSION_MAJOR >= 3)
112 (void)mbedtls_sha256_finish(ctx, sha256);
113#else
114 (void)mbedtls_sha256_finish_ret(ctx, sha256);
115#endif
116 memcpy(output, sha256, obits / 8);
117 }
118}
