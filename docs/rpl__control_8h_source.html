109/* Descriptor for a route from a DIO Route Information option.
110 * Used to hold the "master copy" in the DODAG structure - the table for the
111 * current DODAG is used to populate routes in our system routing table, and to
112 * form our own DIOs.
113 * Not used directly for next hop determination - this is done via the system
114 * routing table, which has per-parent information including parent preferences,
115 * reachability/probing info, etc.
116 */
117typedef struct rpl_dio_route {
118 uint32_t lifetime; /* Lifetime in seconds */
119 ns_list_link_t link;
120 bool age; /* Do we age the lifetime? */
121 uint8_t hold_count; /* Hold count for relaying 0 lifetime */
122 uint8_t flags; /* Flags byte (inc. pref) */
123 uint8_t prefix_len_bytes; /* Prefix length (bytes) - padding preserved when relaying */
124 uint8_t prefix_len; /* Prefix length (bits) */
125 uint8_t prefix[]; /* Variable-length prefix */
126} rpl_dio_route_t;
