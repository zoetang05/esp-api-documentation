34/* Descriptor for a RPL neighbour within a DODAG
35 *
36 * Note that global address is only needed with downward routes, but I don't
37 * think it's worth optimising for an "upward-only" build. (Unless to be a RPL
38 * leaf?)
39 *
40 * DODAG parents are identified by the dodag_parent flag, and they are sorted
41 * first in the instance candidate_neighbour list, in order of preference.
42 */
43struct rpl_neighbour {
44 rpl_dodag_version_t *dodag_version; // Back pointer to DODAG Version
45 uint8_t ll_address[16]; // Link-local address (source of DIO)
46 uint8_t global_address[16]; // Global address (from DIO RIO)
47 bool dodag_parent: 1; // This is a DODAG parent (if true, dodag_version may not be NULL)
48 bool was_dodag_parent: 1; // Was a DODAG parent (used only during parent selection)
49 bool have_global_address: 1; // Global address known
50 bool considered: 1; // Have considered at least once for parent selection
51 bool confirmed: 1; // Confirmed
52 unsigned dodag_pref: 4; // Preference indication for DODAG parents (0=best)
53 uint8_t dao_path_control; // Path control bit assignments for DAO parent
54 uint8_t old_dao_path_control;
55 uint8_t addr_reg_failures; // Address registration failure count (missing ACK)
56 int8_t interface_id;
57 uint8_t g_mop_prf;
58 uint8_t dtsn;
59 uint16_t rank;
60 uint32_t dio_timestamp; // Timestamp of last received DIO message
61 /* Need link quality indication */
62 ns_list_link_t candidate_neighbour_link;
63};
64
65typedef NS_LIST_HEAD(rpl_neighbour_t, candidate_neighbour_link) rpl_neighbour_list_t;

67/* Descriptor for a DODAG Version, identified by (RPLInstanceID, DODAGID, DODAGVersionNumber) */
68struct rpl_dodag_version {
69 rpl_dodag_t *dodag; /* Back pointer to DODAG */
70 uint8_t number; /* Version Number */
71 uint16_t lowest_advertised_rank;
72 uint16_t last_advertised_rank;
73 uint16_t hard_rank_limit; /* Hard rank limit by DAGMaxRankIncrease */
74 uint16_t greediness_rank_limit; /* Rank limit for greediness */
75 ns_list_link_t link;
76};

78/* Descriptor for a DODAG. A DODAG is identified by a (RPLInstanceID, DODAGID) tuple */
79struct rpl_dodag {
80 rpl_instance_t *instance; /* Back pointer to instance */
81 uint32_t timestamp; /* How long since we heard a DIO */
82 uint8_t id[16]; /* Root identifier */
83 uint8_t g_mop_prf; /* Grounded, Mode, Preference */
84 rpl_dodag_conf_int_t config; /* Configuration from DIO */
85 uint8_t info_version; /* Version for g_mop_prf and config */
86 bool root: 1; /* We are the root of this DODAG */
87 bool was_root: 1; /* If we have ever been a root in this DODAG */
88 bool leaf: 1; /* We are a leaf in this DODAG (by policy) */
89 bool have_config: 1; /* We have the config */
90 bool used: 1; /* We have ever been a member of this DODAG? */
91 uint8_t new_config_advertisment_count; /* We have advertiment new config at multicasti DIO max updated value is 0xfe*/
92 NS_LIST_HEAD(rpl_dodag_version_t, link) versions; /* List of DODAG versions (newest first) */
93 prefix_list_t prefixes; /* Prefixes advertised in DIO PIOs */
94 rpl_dio_route_list_t routes; /* Routes advertised in DIO RIOs*/
95 trickle_params_t dio_timer_params; /* Trickle parameters */
96 ns_list_link_t link;
97};

157 } info;

163/* Descriptor for a RPL Instance. An instance can have multiple DODAGs.
164 *
165 * If top bit of instance_id is set then it's a local DODAG, and the dodags
166 * list must have exactly one member, whose dodag_id disambiguates the
167 * instance_id.
168 */
169struct rpl_instance {
170 ns_list_link_t link;
171 rpl_domain_t *domain; /* Back pointer to domain */
172 uint8_t id; /* RPLInstanceID */
173 uint8_t dtsn; /* Our DTSN for this instance */
174 bool neighbours_changed: 1;
175 bool local_repair: 1;
176 bool root_topo_sort_valid: 1;
177 bool root_paths_valid: 1;
178 bool dio_not_consistent: 1; /* Something changed - not consistent this period */
179 bool dao_in_transit: 1; /* If we have a DAO in transit */
180 bool requested_dao_ack: 1; /* If we requested an ACK (so we retry if no ACK, rather than assuming success) */
181 bool pending_neighbour_confirmation: 1; /* if we have not finished address registration state to parent */
182 bool parent_was_selected: 1;
183 bool advertised_dodag_membership_since_last_repair: 1; /* advertised dodag membership since last repair */
184 uint8_t poison_count;
185 uint8_t repair_dis_count;
186 uint16_t repair_dis_timer;
187 uint32_t last_dao_trigger_time;
188 uint16_t srh_error_count; /* SRH errors since last DAO trigger */
189 NS_LIST_HEAD(rpl_dodag_t, link) dodags; /* List of DODAGs */
190 rpl_neighbour_t *wait_response;
191 rpl_neighbour_list_t candidate_neighbours; /* Candidate neighbour set */
192 // rpl_neighbour_list_t old_neighbours; /* Old neighbours (without a live DODAG version) */
193 rpl_dodag_version_t *current_dodag_version; /* Pointer to DODAG version we are a member of (if any) */
194 uint16_t current_rank; /* Current rank in current DODAG Version */
195 uint16_t parent_selection_timer;
196
197 trickle_t dio_timer; /* Trickle timer for DIO transmission */
198 rpl_dao_root_transit_children_list_t root_children;
199 rpl_dao_target_list_t dao_targets; /* List of DAO targets */
200 uint8_t dao_sequence; /* Next DAO sequence to use */
201 uint8_t dao_sequence_in_transit; /* DAO sequence in transit (if dao_in_transit) */
202 uint16_t delay_dao_timer;
203 uint16_t dao_retry_timer;
204 uint8_t dao_attempt;
205 struct rpl_objective *of; /* Objective function pointer */
206};
