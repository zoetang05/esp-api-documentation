383 template<typename EventType, typename IndexType, IndexType MAX_EVENTS>
384 class EventList {
385 public:
386 EventList()
387 {
388 };
389
390 ~EventList()
391 {
392 for (IndexType i = 0; i < _current_size; ++i) {
393 delete _pointers[i];
394 }
395 };
396
402 bool push(EventType *event)
403 {
404 if (_current_size < MAX_EVENTS) {
405 _pointers[_current_size] = event;
406 _current_size++;
407 return true;
408 }
409 return false;
410 };
411
416 EventType* pop()
417 {
418 MBED_ASSERT(_current_size);
419
420 if (!_current_size) {
421 return nullptr;
422 }
423
424 EventType* event_returned = _pointers[_current_index];
425
426 _current_size--;
427 if (_current_size != _current_index) {
428 _pointers[_current_index] = _pointers[_current_size];
429 } else {
430 _current_index = 0;
431 }
432
433 return event_returned;
434 };
435
442 EventType* pop(mbed::Callback<bool(EventType&)> compare_func)
443 {
444 for (IndexType i = 0; i < _current_size ; ++i) {
445 if (compare_func(*_pointers[_current_index])) {
446 return pop();
447 }
448 increment_current_index();
449 }
450
451 return nullptr;
452 }
453
465 EventType* continue_pop(mbed::Callback<bool(EventType&)> compare_func, IndexType *events_not_matching)
466 {
467 _current_index = *events_not_matching;
468 for (IndexType i = *events_not_matching; i < _current_size ; ++i) {
469 if (compare_func(*_pointers[_current_index])) {
470 return pop();
471 }
472 (*events_not_matching)++;
473 increment_current_index();
474 }
475
476 return nullptr;
477 }
478
484 EventType* find(mbed::Callback<bool(EventType&)> compare_func)
485 {
486 for (IndexType i = 0; i < _current_size ; ++i) {
487 if (compare_func(*_pointers[_current_index])) {
488 return _pointers[_current_index];
489 }
490 increment_current_index();
491 }
492
493 return nullptr;
494 }
495
500 IndexType get_size()
501 {
502 return _current_size;
503 }
504
505 private:
506 void increment_current_index()
507 {
508 _current_index++;
509 if (_current_index == _current_size) {
510 _current_index = 0;
511 }
512 }
513
514 private:
515 EventType* _pointers[MAX_EVENTS];
516 IndexType _current_size = 0;
517 /* this helps us find the event faster */
518 IndexType _current_index = 0;
519 };

521#if BLE_FEATURE_PRIVACY && BLE_GAP_HOST_BASED_PRIVATE_ADDRESS_RESOLUTION
522 class PendingAdvertisingReportEvent {
523 public:
524 PendingAdvertisingReportEvent(
525 const AdvertisingReportEvent& event_to_copy
526 ) : event(event_to_copy)
527 {
528 /* copy the data to the buffer */
529 const mbed::Span<const uint8_t> payload = event_to_copy.getPayload();
530 if (payload.size()) {
531 advertising_data_buffer = new(std::nothrow) uint8_t[payload.size()];
532 if (advertising_data_buffer) {
533 memcpy(advertising_data_buffer, payload.data(), payload.size());
534 /* set the payload to our local copy of the data */
535 event.setAdvertisingData(mbed::make_Span(advertising_data_buffer, payload.size()));
536 }
537 }
538 };
539
540 ~PendingAdvertisingReportEvent()
541 {
542 delete[] advertising_data_buffer;
543 }
544
545 bool is_valid()
546 {
547 return advertising_data_buffer || (event.getPayload().size() == 0);
548 }
549
550 AdvertisingReportEvent& get_pending_event()
551 {
552 return event;
553 }
554
555 private:
556 AdvertisingReportEvent event;
557 uint8_t *advertising_data_buffer = nullptr;
558 };
559#endif // BLE_FEATURE_PRIVACY && BLE_GAP_HOST_BASED_PRIVATE_ADDRESS_RESOLUTION
