147/* General macros to handle TLV type */
148#define MAJOR_MASK 0xF /* 4 bit */
149#define MAJOR_POS 12 /* 12 bit */
150#define MINOR_MASK 0xFFF /* 12 bit */

152#define SET_TLV_TYPE(major, minor) \
153 ((((major) & MAJOR_MASK) << MAJOR_POS) | ((minor) & MINOR_MASK))
154#define GET_MAJOR(tlv_type) ((tlv_type) >> MAJOR_POS)
155#define GET_MINOR(tlv_type) ((tlv_type) & MINOR_MASK)
