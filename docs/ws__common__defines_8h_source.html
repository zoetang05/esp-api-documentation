524/* WS requires at least 19 MAC retransmissions (total 1+19=20 attempts). Default 802.15.4 macMaxFrameRetries is 3 (total 1+3=4 attempts).
525 * At least 4 request restarts must be used: (Initial channel + WS_TX_REQUEST_RESTART_MAX) * MAC attempts = (1+4)*4=20 attempts
526 *
527 * Valid settings could be for example:
528 * WS_MAX_FRAME_RETRIES WS_TX_REQUEST_RESTART_MAX Total attempts
529 * 0 19 1+0*1+19=20
530 * 1 9 1+1*1+9=20
531 * 2 6 1+2*1+6=21
532 * 3 4 1+3*1+4=20
533 *
534 */
535// This configuration is used when bootstrap is ready
536#define WS_MAX_FRAME_RETRIES 3
537// This configuration is used during bootstrap
538#define WS_MAX_FRAME_RETRIES_BOOTSTRAP 0
539
540// Configuring data request restart allows MAC to push failed packet back to MAC TX queue up to WS_CCA_REQUEST_RESTART_MAX times for CCA failure and WS_TX_REQUEST_RESTART_MAX for TX failure.
541// Packet cannot be taken back to transmission before it has finished the blacklist period.
542#define WS_CCA_REQUEST_RESTART_MAX 9
543#define WS_TX_REQUEST_RESTART_MAX 4
544#define WS_TX_REQUEST_RESTART_MAX_BOOTSTRAP 19
545#define WS_REQUEST_RESTART_BLACKLIST_MIN 20
546#define WS_REQUEST_RESTART_BLACKLIST_MAX 300
547
548#if (1 + WS_MAX_FRAME_RETRIES) * (1 + WS_TX_REQUEST_RESTART_MAX) < 20
549#warning "MAX frame retries set too low"
550#endif
