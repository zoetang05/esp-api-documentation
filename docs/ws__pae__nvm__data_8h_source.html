32#define PAE_NVM_NW_INFO_TAG 1
33#define PAE_NVM_KEYS_TAG 2
34#define PAE_NVM_FRAME_COUNTER_TAG 3
35#define PAE_NVM_KEY_STORAGE_INDEX_TAG 4
36#define PAE_NVM_KEY_STORAGE_TAG 5

38// pan_id (2) + network name (33) + GTK EUI-64 (own EUI-64) (8) + stored time (8) + time changed (1) + (GTK set (1) + GTK expiry timestamp (8) + status (1) + install order (1) + GTK (16)) * 4
39#define PAE_NVM_NW_INFO_LEN 2 + 33 + 8 + 8 + 1 + (1 + 8 + 1 + 1 + GTK_LEN) * GTK_NUM
40
41// PTK EUI-64 set (1) + PTK EUI-64 (8) + PMK set (1) + PMK lifetime (4) + PMK (32) + PMK replay counter (8) + PTK set (1) + PTK lifetime (4) + PTK (48)
42#define PAE_NVM_KEYS_LEN 1 + 8 + 1 + 4 + PMK_LEN + 8 + 1 + 4 + PTK_LEN
43
44// restart counter + stored time + pan version + (frame counter set (1) + GTK (16) + frame counter (4) + max frame counter change (4)) * 4
45#define PAE_NVM_FRAME_COUNTER_LEN 4 + 8 + 2 + (1 + GTK_LEN + 4 + 4) * GTK_NUM
46
47#if (PAE_NVM_NW_INFO_LEN >= PAE_NVM_KEYS_LEN) && (PAE_NVM_NW_INFO_LEN >= PAE_NVM_FRAME_COUNTER_LEN)
48#define PAE_NVM_LARGEST_FILE PAE_NVM_NW_INFO_LEN
49#elif (PAE_NVM_KEYS_LEN >= PAE_NVM_NW_INFO_LEN) && (PAE_NVM_KEYS_LEN >= PAE_NVM_FRAME_COUNTER_LEN)
50#define PAE_NVM_LARGEST_FILE PAE_NVM_KEYS_LEN
51#else
52#define PAE_NVM_LARGEST_FILE PAE_NVM_FRAME_COUNTER_LEN
53#endif
